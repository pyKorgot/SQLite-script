create table users (
id integer NOT NULL primary key AUTOINCREMENT,
second_name text,
first_name text,
patronymic text,
region_id integer,
city_id integer,
phone text,
email text,
FOREIGN KEY (city_id) REFERENCES cities(id),
FOREIGN KEY (region_id) REFERENCES regions(id)
);


create table regions(
id integer not null primary key AUTOINCREMENT,
region_name text);
